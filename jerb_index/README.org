* Installation
  # Install deps
  sudo apt-get install redis-server
  pip3 install flask
  pip3 install flask_restful --user
  pip3 install redis --user

  # Setup local environment variables
  export REDIS_HOST='localhost'
  export REDIS_PORT='6379'
  export REDIS_PASS=''"

  # Launch
  cd nems
  python3 -m jerbs_index
